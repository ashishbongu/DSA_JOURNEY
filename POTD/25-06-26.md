# 3737. Count Subarrays With Majority Element I

### MY INITIAL APPRAOCH

```
class Solution {
    public int countMajoritySubarrays(int[] nums, int target) {
        
        int count =0;


        for(int i =0;i<nums.length;i++){
            for(int j=i;j<nums.length;j++){

                    int[] freq = new int[10];
                    for(int k=i;k<=j;k++){
                        freq[nums[k]]++;
                    }

                    int max = Integer.MIN_VALUE;
                    int temp;
                    int max_pos=0;
                    for(int k=0;k<freq.length;k++){
                        temp = max;
                        max=Math.max(max,freq[k]);
                        if(temp!=max){
                            max_pos=k;
                        }
                    }

                    if(max>(j-i+1)/2 && max_pos==target){
                        count++;
                    }
            }
        }

        return count;
    }
}

### updated with hash map since there can be numbers above 9 (but encountered TLE)

```
class Solution {
    public int countMajoritySubarrays(int[] nums, int target) {
        
        int count =0;


        for(int i =0;i<nums.length;i++){
            for(int j=i;j<nums.length;j++){

                    HashMap<Integer,Integer> freq = new HashMap<>();
                    for(int k=i;k<=j;k++){
                        if (freq.containsKey(nums[k])) {
                        freq.put(nums[k], freq.get(nums[k]) + 1);
                        } else {
                            freq.put(nums[k], 1);
                        }
                    }

                    int maxKey = -1;
                    int maxValue = Integer.MIN_VALUE;

                    for (Map.Entry<Integer, Integer> entry : freq.entrySet()) {
                        if (entry.getValue() > maxValue) {
                            maxValue = entry.getValue();
                            maxKey = entry.getKey();
                        }
                    }


                    if(maxValue>(j-i+1)/2 && maxKey==target){
                        count++;
                    }
            }
        }

        return count;
    }
}
